Example 5c: overbinding with equality

for arbitrary a, b:

$\Sigma = \{P(a), a=b\}$

$\Pi = \{\lnot P(b)\}$

\begin{prooftree}
	\AxiomCm{\sig{P(a)}}
	\AxiomCm{\sig{a=b}}
	\BinaryInfCm{P(b)}

	\AxiomCm{\p{\lnot P(b)}}
	\BinaryInfCm{\square}
\end{prooftree}

\begin{prooftree}
	\AxiomCm{\bot}
	\AxiomCm{\bot}
	\BinaryInfCm{\bot}

	\AxiomCm{\top}
	\BinaryInfCm{P(b)}
\end{prooftree}
